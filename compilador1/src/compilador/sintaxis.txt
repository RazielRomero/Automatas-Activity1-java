terminal DISTINTO, ASISGNADOR, IGUAL, MENOSNUMERO, MENORNUMERO,
         MASNUMERO, MAYORIGUAL, MASUNO, MENOSUNO, DIVISION, DIVIDIDONUMERO,
         POR, PORNUMERO, MAS, MENOS, MAYOR, MENOR;

terminal IDENTIFICADOR, LOG, DOBLE, IF, ELSE, WHILE, BREAK, CONTINUE,
         READ, WRITE, PARIZQ, PARDER, LLAVEIZQ, LLAVEDER. PYCOMA, COMA, AND, OR;

terminal long numero NUMERO_LONG;

terminal double NUMERO_DOUBLE;

non terminal SECUENCIA, ORACION, INSTRUCCION, DECLARACION,
             ASISGNACION, SI, SINO, CONDICION, MIENTRAS, LEER, ESCRIBIR, CONTROL, 
             COMPARADOR, NUMERO, BLOQUEBUCLE;

non terminal OY, TERMINO, OPERADOR, LISTADO, OPERANDO,
             TIPO, SIBUCLE, SINOBUCLE;

SECUENCIA : := OPERACION SECUENCIA
            | SI SECUENCIA
            | MIENTRAS SECUENCIA
            |
            ;|

OPERACION : :=  INSTRUCCION PYCOMA
            | PYCOMA
            ;


INSTRUCCION : := DECLARACION
                | ASISGNACION
                | LEER
                | ESCRIBIR
                ;

DECLARACION : := TIPO IDENTIFICADOR ASISGNADOR TERMINO
                | TIPO LISTADO
                ;

ASISGNADOR : := IDENTIFICADOR ASISGNADOR TERMINO
                | IDENTIFICADOR MASUNO
                | IDENTIFICADOR MENOSUNO
                | IDENTIFICADOR MENOSNUMERO TERMINO
                | IDENTIFICADOR MASNUMERO TERMINO
                | IDENTIFICADOR PORNUMERO TERMINO
                | IDENTIFICADOR DIVIDIDONUMERO TERMINO
                ;|

SI : := IF PARIZQ CONDICION PARDER OPERACION
    | IF PARIZQ CONDICION PARDER LLAVEIZQ SECUENCIA LLAVEDER
    | IF PARIZQ CONDICION PARDER OPERACION SINO
    | IF PARIZQ CONDICION PARDER LLAVEIZQ SECUENCIA LLAVEDER SINO
    ;

SINO : := ELSE OPERACION
        |  ELSE LLAVEIZQ SECUENCIA LLAVEDER
        ;|

MIENTRAS : := WHILE PARIZQ CONDICION PARDER OPERACION
            | WHILE PARIZQ CONDICION PARDER CONTROL
            | WHILE PARIZQ CONDICION PARDER LLAVEIZQ BLOQUEBUCLE LLAVEDER
            ;

CONTROL ::= BREAK PYCOMA
    | CONTINUE PYCOMA
    ;

CONDICION ::= TERMINO COMPARADOR TERMINO
            | CONDICION OY CONDICION
            | PARIZQ CONDICION PARDER
            ;

NUMERO ::= NUMERO_DOUBLE
            | NUMERO_LONG

BLOQUEBUCLE ::= CONTROL BLOQUEBUCLE
            | ORACION BLOQUEBUCLE
            | SIBUCLE BLOQUEBUCLE
            | MIENTRAS BLOQUEBUCLE
            |
            ;

OY ::= AND
        | OR
        ;

TERMINO ::= OPERANDO
        | OPERANDO OPERADOR TERMINO
        | PARIZQ TERMINO PARDER
        ;

OPERADOR ::= MAS
        | MENOS
        | POR
        | DIVISION
        ;

LISTADO ::= IDENTIFICADOR
        | IDENTIFICADOR COMA LISTADO

COMPARADOR ::= IGUAL
        | DISTINTO
        | MAYORIGUAL
        | MENORIGUAL
        | MAYOR
        | MENOR
        ;

LEER ::= READ IDENTIFICADOR
        ;

ESCRIBIR ::= NUMERO
        | IDENTIFICADOR
        ;

TIPO ::= DOUBLE
        | LONG
        ;

SIBUCLE ::= IF PARIZQ CONDICION PARDER ORACION
        | IF PARIZQ CONDICION PARDER CONTROL
        | IF PARIZQ CONDICION PARDER LLAVEIZQ BLOQUEBUCLE LLAVEDER
        | IF PARIZQ CONDICION PARDER ORACION SINO
        | IF PARIZQ CONDICION PARDER CONTROL SINO
        | IF PARIZQ CONDICION PARDER LLAVEIZQ BLOQUEBUCLE LLAVEDER SINO
        ;

SINOBUCLE ::= ELSE ORACION
        | ELSE CONTROL
        | ELSE LLAVEIZQ BLOQUEBUCLE LLAVEDER
        ;
